# 手写Promise简单版

​ Promise是解决异步回调问题的解决方式，关于Promise的使用，可以参考阮一峰老师的ES6教程，这里仅对Promise实现做一个简单版的手写。

class Promise() {
  //构造器
  consructor(exec) {
    //Promise有三个状态pending（等待）、fulfilled（完成）、rejected（错误）
    this.state = "pending";//一开始为等待状态
    this.resValue = "";//成功返回的值
    this.rejValue = "";//报错信息
    this.onResolveCallBacks = []; //解决异步代码，将成功后需要调用的函数存在里面
    this.onRejectCallBacks = [];//同理
    //成功函数
    let resolve = (value) => {
      if (this.state === "pending") {
        this.state = "fulfilled";
        this.resValue = value;
        this.onResolvedCallbacks.forEach(fn=>fn());
      }    
    };
    //失败函数
    let reject = (value) => {
      if (this.state === "pending") {
        this.state = "rejected";
        this.rejValue = value;
        this.onRejectedCallbacks.forEach(fn=>fn());
      }
    };
    try {
      exec(resolve, reject);  
    } catch(err) {
      reject(err);
    }
  };
  //Promise还需要一个then方法
  then(onFulfilled, onRejected) {
    if (this.state === "fufilled") {
      onFullfilled(this.resValue);
    } else if (this.state === "rejected") {
      onRejected(this.rejValue);
    } else if (this.state === "pending") {
      this.onResolveCallBacks.push(() => {
        onFulfilled(this.resValue);
      });
      this.onRejectedCallBacks.push(() => {
        onRejected(this.rejValue);
      })
    };
  }  
}

​ 以上方法可以实现同步、异步的Promise，但是无法实现then的链式调用，是最简单的Promise版本。
